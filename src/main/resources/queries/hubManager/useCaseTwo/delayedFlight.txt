// flight0 is delayed

MATCH 
	(f:flight{from:'Paris'})-->(p:passenger)-->(i:itinerary)-->(leg:flight)//-->(p1:passenger)-->(i1:itinerary)
WHERE leg.to = f.from AND (leg.ETD - f.ETA) < 45*60
WITH p
MATCH (p:passenger)-->(i:itinerary)-->(person:person)
return 
p.flightId as flight,
person.name as name,
person.surname as surname
